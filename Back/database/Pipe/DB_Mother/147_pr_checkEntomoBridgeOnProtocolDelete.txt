Create PROCEDURE [dbo].[pr_checkEntomoBridgeOnProtocolDelete]
 @obsID int,
 @result int OUTPUT
   AS
   BEGIN

   SET NOCOUNT ON
	DECLARE
	@nbChild int


	SELECT @nbChild = count(*)
	FROM Observation o
	WHERE Parent_Observation = @obsID
	AND original_id like '%ecollection%'

	IF @nbChild > 0
		SELECT @result
	
	END


 GO

GO
