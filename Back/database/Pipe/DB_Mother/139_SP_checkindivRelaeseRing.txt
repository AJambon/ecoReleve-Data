CREATE PROCEDURE [dbo].[pr_Check_Existing_Indiv_ReleaseRing]

@code varchar(250), 
@id_indiv int,
@result int OUTPUT

AS 
SET NOCOUNT ON

IF @id_indiv is null
SET @id_indiv = 0	

SELECT @result = (
	SELECT 1 WHERE EXISTS (
	SELECT *
  FROM [IndividualDynPropValuesNow] v
  where name = 'Release_Ring_Code'
  AND ValueString = @code
  and FK_Individual != @id_indiv
  )
  )

 select @result



INSERT INTO [dbo].[TVersion] (TVer_FileName,TVer_Date,TVer_DbName) VALUES ('139_SP_checkindivRelaeseRing',GETDATE(),(SELECT db_name()))


GO
